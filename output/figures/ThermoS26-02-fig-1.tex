\documentclass[tikz,border=10pt]{standalone}
\usepackage{tikz}
\usepackage{tikz-3dplot}
\usetikzlibrary{arrows.meta,calc}

\begin{document}
\tdplotsetmaincoords{70}{110}
\begin{tikzpicture}[tdplot_main_coords,scale=1.5]

    % Define origin
    \coordinate (O) at (0,0,0);

    % Define axis lengths
    \def\axislen{3.5}

    % Draw axes with professional styling
    % V axis (pointing right in the diagram)
    \draw[-{Stealth[length=3mm,width=2mm]},thick] (O) -- (\axislen,0,0) node[anchor=north west,font=\Large] {$V$};

    % N axis (pointing up)
    \draw[-{Stealth[length=3mm,width=2mm]},thick] (O) -- (0,0,\axislen) node[anchor=south,font=\Large] {$N$};

    % S axis (pointing down-left in 3D perspective)
    \draw[-{Stealth[length=3mm,width=2mm]},thick] (O) -- (0,\axislen,0) node[anchor=north east,font=\Large] {$S$};

    % Add origin marker
    \fill (O) circle (1.5pt);

    % Optional: Add subtle grid planes to show 3D structure
    \begin{scope}[opacity=0.15]
        % VS plane
        \draw[gray] (0,0,0) -- (0,2.5,0) -- (0,2.5,2.5) -- (0,0,2.5) -- cycle;
        % VN plane
        \draw[gray] (0,0,0) -- (2.5,0,0) -- (2.5,0,2.5) -- (0,0,2.5) -- cycle;
        % SN plane
        \draw[gray] (0,0,0) -- (0,2.5,0) -- (2.5,2.5,0) -- (2.5,0,0) -- cycle;
    \end{scope}

    % Add subtle tick marks on axes
    \foreach \x in {1,2,3} {
        \draw[thin,gray] (\x,0,0) -- (\x,-0.08,0);
        \draw[thin,gray] (0,\x,0) -- (-0.08,\x,0);
        \draw[thin,gray] (0,0,\x) -- (0,-0.08,\x);
    }

\end{tikzpicture}
\end{document}
